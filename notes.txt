query for getting data

SELECT
  DATE,
  SourceCommonName,
  DocumentIdentifier,
  V1LOCATIONS as Locations,
  V1ORGANIZATIONS as Organizations
FROM `gdelt-bq.gdeltv2.gkg`
WHERE DATE >= 20250123000000
  AND SourceCommonName IN (
    'washingtonpost.com',
    'nytimes.com',
    'foxnews.com',
    'usatoday.com',
    'wsj.com',
    'apnews.com',
    'nbcnews.com',
    'npr.org',
    'breitbart.com',
    'cnn.com'
  )
  AND (
    Organizations LIKE '%IMMIGRATION AND CUSTOMS ENFORCEMENT%'
    OR Organizations LIKE '%US IMMIGRATION AND CUSTOMS%'
  )
LIMIT 1000
